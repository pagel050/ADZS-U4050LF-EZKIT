156 ADI_TMR_RESULT adi_tmr_Init(ADI_TMR_DEVICE const eDevice, ADI_CALLBACK const pfCallback, void * const pCBParam, bool bEnableInt) {
157 #ifdef ADI_DEBUG
158  /* IF(Bad device input parameter) */
159  if (eDevice >= ADI_TMR_DEVICE_NUM) {
160  return ADI_TMR_BAD_DEVICE_NUM;
161  } /* ENDIF */
162  /* IF(The timer is already running) */
163  if ((adi_tmr_registers[eDevice]->CTL & BITM_TMR_RGB_CTL_EN) == BITM_TMR_RGB_CTL_EN) {
164  return ADI_TMR_OPERATION_NOT_ALLOWED;
165  } /* ENDIF */
166 #endif
167  /* Setup the callback function */
168  adi_tmr_callbacks [eDevice] = pfCallback;
169  adi_tmr_parameters[eDevice] = pCBParam;
170 
171  /* IF(Enable interrupt) */
172  if (bEnableInt == true) {
173  NVIC_EnableIRQ(adi_tmr_interrupt[eDevice]);
174  /* ELSE(Disable interrupt) */
175  } else {
176  NVIC_DisableIRQ(adi_tmr_interrupt[eDevice]);
177  } /* ENDIF */
178 
179  /* Static configuration */
180 #if ADI_TIMER_ENABLE_STATIC_CONFIG_SUPPORT == 1u
181  adi_tmr_registers[eDevice]->CTL = aTimerCtlConfig [eDevice];
182  adi_tmr_registers[eDevice]->LOAD = aTimerLoadConfig [eDevice];
183  adi_tmr_registers[eDevice]->ALOAD = aTimerALoadConfig [eDevice];
184  adi_tmr_registers[eDevice]->EVENTSELECT = aTimerEventConfig [eDevice];
185  adi_tmr_registers[eDevice]->PWM0CTL = aTimerPwmCtlConfig [eDevice];
186  adi_tmr_registers[eDevice]->PWM0MATCH = aTimerPwmMatchConfig[eDevice];
187 
188  /* IF(Initializing the RGB timer, there are 2 other PWM outputs to configure) */
189  if (eDevice == ADI_TMR_DEVICE_RGB) {
190  /* The array is bumped by 1 to get to the 5th entry in the static config array, which contains RGB PWM1 */
191  adi_tmr_registers[eDevice]->PWM1CTL = aTimerPwmCtlConfig [eDevice+1u];
192  adi_tmr_registers[eDevice]->PWM1MATCH = aTimerPwmMatchConfig[eDevice+1u];
193  /* The array is bumped by 2 to get to the 6th entry in the static config array, which contains RGB PWM2 */
194  adi_tmr_registers[eDevice]->PWM2CTL = aTimerPwmCtlConfig [eDevice+2u];
195  adi_tmr_registers[eDevice]->PWM2MATCH = aTimerPwmMatchConfig[eDevice+2u];
196  } /* ENDIF */
197 #endif
198 
199  return ADI_TMR_SUCCESS;
200 }

222 ADI_TMR_RESULT adi_tmr_ConfigTimer(ADI_TMR_DEVICE const eDevice, ADI_TMR_CONFIG timerConfig) {
223  uint16_t nTemp;
224 #ifdef ADI_DEBUG
225  /* IF(Bad device input parameter) */
226  if (eDevice >= ADI_TMR_DEVICE_NUM) {
227  return ADI_TMR_BAD_DEVICE_NUM;
228  } /* ENDIF */
229  /* IF(Bad configuration, cannot enable reloading while in free running mode) */
230  if ((timerConfig.bPeriodic == false) && (timerConfig.bReloading == true)) {
231  return ADI_TMR_BAD_RELOAD_CONFIGURATION;
232  } /* ENDIF */
233  /* IF(The timer is already running) */
234  if ((adi_tmr_registers[eDevice]->CTL & BITM_TMR_RGB_CTL_EN) == BITM_TMR_RGB_CTL_EN) {
235  return ADI_TMR_OPERATION_NOT_ALLOWED;
236  } /* ENDIF */
237 #endif
238  /* Set the load registers */
239  adi_tmr_registers[eDevice]->LOAD = timerConfig.nLoad;
240  adi_tmr_registers[eDevice]->ALOAD = timerConfig.nAsyncLoad;
241 
242  /* IF(Busy bit does not clear after waiting) */
243  if (ADI_TMR_SUCCESS != WaitForStatusBit(eDevice, (uint16_t) BITM_TMR_RGB_STAT_BUSY)) {
244  return ADI_TMR_DEVICE_BUSY;
245  } /* ENDIF */
246 
247  /* Read the control register and clear everything aside to the event capture bits, which are the only fields not set in this function */
248  nTemp = adi_tmr_registers[eDevice]->CTL;
249  nTemp &= (uint16_t) (BITM_TMR_RGB_CTL_EVTEN | BITM_TMR_RGB_CTL_RSTEN);
250 
251  /* Setup the prescaler and the clock source */
252  nTemp |= (uint16_t)(((uint16_t) timerConfig.ePrescaler ) << BITP_TMR_RGB_CTL_PRE);
253  nTemp |= (uint16_t)(((uint16_t) timerConfig.eClockSource) << BITP_TMR_RGB_CTL_CLK);
254 
255  /* IF(Periodic mode) */
256  if (timerConfig.bPeriodic == true) {
257  nTemp |= (1u << BITP_TMR_RGB_CTL_MODE);
258  } /* ENDIF */
259 
260  /* IF(Counting up) */
261  if (timerConfig.bCountingUp == true) {
262  nTemp |= (1u << BITP_TMR_RGB_CTL_UP);
263  } /* ENDIF */
264 
265  /* IF(Reloading is enabled) */
266  if (timerConfig.bReloading == true) {
267  nTemp |= (1u << BITP_TMR_RGB_CTL_RLD);
268  } /* ENDIF */
269 
270  /* IF(Sync bypass is enabled) */
271  if (timerConfig.bSyncBypass == true) {
272  nTemp |= (1u << BITP_TMR_RGB_CTL_SYNCBYP);
273  } /* ENDIF */
274 
275  /* Update the control register with the new configuration */
276  adi_tmr_registers[eDevice]->CTL = nTemp;
277 
278  return ADI_TMR_SUCCESS;
279 }

304 ADI_TMR_RESULT adi_tmr_ConfigEvent(ADI_TMR_DEVICE const eDevice, ADI_TMR_EVENT_CONFIG eventConfig) {
305 #ifdef ADI_DEBUG
306  /* IF(Bad device input parameter) */
307  if (eDevice >= ADI_TMR_DEVICE_NUM) {
308  return ADI_TMR_BAD_DEVICE_NUM;
309  } /* ENDIF */
310  /* IF(Bad event input parameter) */
311  if (eventConfig.nEventID >= ADI_TMR_NUM_EVENTS) {
312  return ADI_TMR_BAD_EVENT_ID;
313  } /* ENDIF */
314  /* IF(The timer is already running) */
315  if ((adi_tmr_registers[eDevice]->CTL & BITM_TMR_RGB_CTL_EN) == BITM_TMR_RGB_CTL_EN) {
316  return ADI_TMR_OPERATION_NOT_ALLOWED;
317  } /* ENDIF */
318 #endif
319  /* Set the event number */
320  adi_tmr_registers[eDevice]->EVENTSELECT = (uint16_t) eventConfig.nEventID;
321 
322  /* IF(Busy bit does not clear after waiting) */
323  if (ADI_TMR_SUCCESS != WaitForStatusBit(eDevice, (uint16_t) BITM_TMR_RGB_STAT_BUSY)) {
324  return ADI_TMR_DEVICE_BUSY;
325  } /* ENDIF */
326 
327  /* Clear the event enable bit and keep the other bits */
328  adi_tmr_registers[eDevice]->CTL &= (uint16_t) ~(BITM_TMR_RGB_CTL_EVTEN | BITM_TMR_RGB_CTL_RSTEN);
329 
330  /* IF(Turning event capture on) */
331  if (eventConfig.bEnable == true) {
332  adi_tmr_registers[eDevice]->CTL |= (uint16_t) BITM_TMR_RGB_CTL_EVTEN;
333  } /* ENDIF */
334 
335  /* IF(Enabling reset on event capture) */
336  if (eventConfig.bPrescaleReset == true) {
337  adi_tmr_registers[eDevice]->CTL |= (uint16_t) BITM_TMR_RGB_CTL_RSTEN;
338  } /* ENDIF */
339 
340  return ADI_TMR_SUCCESS;
341 }

367 ADI_TMR_RESULT adi_tmr_ConfigPwm(ADI_TMR_DEVICE const eDevice, ADI_TMR_PWM_CONFIG pwmConfig) {
368  uint16_t nControl = 0u;
369 #ifdef ADI_DEBUG
370  /* IF(Bad device input parameter) */
371  if (eDevice >= ADI_TMR_DEVICE_NUM) {
372  return ADI_TMR_BAD_DEVICE_NUM;
373  } /* ENDIF */
374  /* IF(The timer is already running) */
375  if ((adi_tmr_registers[eDevice]->CTL & BITM_TMR_RGB_CTL_EN) == BITM_TMR_RGB_CTL_EN) {
376  return ADI_TMR_OPERATION_NOT_ALLOWED;
377  } /* ENDIF */
378  /* IF(Bad PWM output and device combo OR bad PWM output) */
379  if (((eDevice != ADI_TMR_DEVICE_RGB) && (pwmConfig.eOutput != ADI_TMR_PWM_OUTPUT_0)) || (pwmConfig.eOutput >= ADI_TMR_PWM_OUTPUT_NUM)) {
380  return ADI_TMR_BAD_PWM_NUM;
381  } /* ENDIF */
382 #endif
383  /* IF(Idle high is set) */
384  if (pwmConfig.bIdleHigh == true) {
385  nControl = (1u << ((uint16_t) BITP_TMR_RGB_PWM0CTL_IDLESTATE));
386  } /* ENDIF */
387 
388  /* IF(Match mode is enabled) */
389  if (pwmConfig.bMatch == true) {
390  nControl |= (1u << ((uint16_t) BITP_TMR_RGB_PWM0CTL_MATCH));
391  } /* ENDIF */
392 
393  /* IF(PWM output 0) */
394  if (pwmConfig.eOutput == ADI_TMR_PWM_OUTPUT_0) {
395  adi_tmr_registers[eDevice]->PWM0CTL = nControl;
396  adi_tmr_registers[eDevice]->PWM0MATCH = pwmConfig.nMatchValue;
397  /* IF(PWM output 1) */
398  } else if (pwmConfig.eOutput == ADI_TMR_PWM_OUTPUT_1) {
399  adi_tmr_registers[eDevice]->PWM1CTL = nControl;
400  adi_tmr_registers[eDevice]->PWM1MATCH = pwmConfig.nMatchValue;
401  /* ELSE(PWM output 2) */
402  } else {
403  adi_tmr_registers[eDevice]->PWM2CTL = nControl;
404  adi_tmr_registers[eDevice]->PWM2MATCH = pwmConfig.nMatchValue;
405  } /* ENDIF */
406 
407  return ADI_TMR_SUCCESS;
408 }

556 static ADI_TMR_RESULT WaitForStatusBit(ADI_TMR_DEVICE const eDevice, uint16_t nBusyBit) {
557  /* FOR(Number of arbitrary iterations) */
558  for (uint16_t i = 0u; i < 1000u; i++) {
559  /* IF(Busy bit is low) */
560  if ((adi_tmr_registers[(eDevice)]->STAT & nBusyBit) == ((uint16_t) 0u)) {
561  return ADI_TMR_SUCCESS;
562  } /* ENDIF */
563  } /* ENDFOR */
564  return ADI_TMR_DEVICE_BUSY;
565 }

567 static void CommonIntHandler(ADI_TMR_DEVICE const eDevice) {
568  /* Read status register */
569  uint16_t IntStatus = adi_tmr_registers[eDevice]->STAT;
570  /* IF(Callback function has been set) */
571  if(adi_tmr_callbacks[eDevice] != NULL) {
572  /* IF(Timeout interrupt occurred) */
573  if((IntStatus & ((uint16_t) BITM_TMR_RGB_STAT_TIMEOUT)) != ((uint16_t) 0u)) {
574  adi_tmr_callbacks[eDevice](adi_tmr_parameters[eDevice], ADI_TMR_EVENT_TIMEOUT, NULL);
575  } /* ENDIF */
576  /* IF(Event capture interrupt occurred) */
577  if((IntStatus & ((uint16_t) BITM_TMR_RGB_STAT_CAPTURE)) != ((uint16_t) 0u)) {
578  adi_tmr_callbacks[eDevice](adi_tmr_parameters[eDevice], ADI_TMR_EVENT_CAPTURE, NULL);
579  } /* ENDIF */
580  } /* ENDIF */
581  /* Clear pending interrupt */
582  adi_tmr_registers[eDevice]->CLRINT = (BITM_TMR_RGB_CLRINT_EVTCAPT | BITM_TMR_RGB_CLRINT_TIMEOUT);
583 }

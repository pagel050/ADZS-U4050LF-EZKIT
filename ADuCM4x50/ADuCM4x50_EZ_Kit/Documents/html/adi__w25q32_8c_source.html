133 ADI_W25Q32_RESULT adi_w25q32_Open(void *pDevMemory, uint32_t nMemorySize, ADI_W25Q32_HANDLE * phDevice)
134 {
135  ADI_W25Q32_HANDLE hDevice;
136  ADI_SPI_RESULT eSpiResult;
137  uint8_t rxBuffer[3u];
138 
139 #ifdef ADI_DEBUG
140  if (nMemorySize != sizeof(struct __ADI_W25Q32_DATA_TYPE))
141  {
142  return ADI_W25Q32_FAILED_INSUFFICIENT_MEMORY;
143  }
144 #endif
145 
146  /* Clear the user provided memory */
147  memset(pDevMemory, 0, nMemorySize);
148 
149  /* Map the user provided memory to the driver data structure */
150  hDevice = pDevMemory;
151 
152  eSpiResult = adi_spi_Open(ADI_W25Q32_CFG_SPI_DEVICE_NUM, hDevice->aDeviceMemory, ADI_SPI_MEMORY_SIZE, &(hDevice->hSpiDevice));
153  RETURN_ERROR(eSpiResult, ADI_SPI_SUCCESS, ADI_W25Q32_FAILED_SPI_ERROR);
154 
155  eSpiResult = adi_spi_SetChipSelect(hDevice->hSpiDevice, ADI_W25Q32_CFG_SPI_CS_NUM);
156  RETURN_ERROR(eSpiResult, ADI_SPI_SUCCESS, ADI_W25Q32_FAILED_SPI_ERROR);
157 
158  /* Check the device and manf id */
159  command(hDevice, ADI_W25Q32_CMD_ID, rxBuffer, 3u);
160  RETURN_ERROR(rxBuffer[0u], ADI_W25Q32_MANF_ID, ADI_W25Q32_FAILED_BAD_PART_INFO);
161  RETURN_ERROR(rxBuffer[1u], ADI_W25Q32_DEVI_ID_H, ADI_W25Q32_FAILED_BAD_PART_INFO);
162  RETURN_ERROR(rxBuffer[2u], ADI_W25Q32_DEVI_ID_L, ADI_W25Q32_FAILED_BAD_PART_INFO);
163 
164  /* Make the callers handle map to the device data structure */
165  *phDevice = hDevice;
166 
167  return ADI_W25Q32_SUCCESS;
168 }

219 ADI_W25Q32_RESULT adi_w25q32_Write(ADI_W25Q32_HANDLE hDevice, uint8_t * buf, uint32_t bufSize, uint32_t address)
220 {
221  ADI_SPI_TRANSCEIVER transceive;
222  ADI_W25Q32_RESULT eResult;
223  ADI_SPI_RESULT eSpiResult;
224  uint32_t nHwError;
225 
226 #ifdef ADI_DEBUG
227  /* Verify no buffer overflow */
228  if (bufSize > ADI_W25Q32_PAGE_SIZE)
229  {
230  return ADI_W25Q32_FAILED_BAD_BUFFER_SIZE;
231  }
232 
233  /* Verify valid address */
234  if (address > (ADI_W25Q32_PAGE_SIZE*ADI_W25Q32_NUM_PAGES))
235  {
236  return ADI_W25Q32_FAILED_BAD_ADDRESS;
237  }
238 #endif
239 
240  /* Turn on write enable */
241  eResult = command(hDevice, ADI_W25Q32_CMD_WENAB, NULL, 0u);
242  RETURN_ERROR(eResult, ADI_W25Q32_SUCCESS, eResult);
243  eResult = busyWait(hDevice);
244  RETURN_ERROR(eResult, ADI_W25Q32_SUCCESS, eResult);
245 
246  transceive.TransmitterBytes = (uint16_t) (ADI_W25Q32_PROLOGUE + bufSize);
247  transceive.ReceiverBytes = 0u;
248  transceive.nTxIncrement = 1u;
249  transceive.nRxIncrement = 1u;
250  transceive.bDMA = true;
251  transceive.bRD_CTL = false;
252  transceive.pTransmitter = hDevice->aTransmit;
253  transceive.pReceiver = NULL;
254 
255  /* Fill TX buffer */
256  hDevice->aTransmit[0u] = ADI_W25Q32_CMD_WRITE;
257  hDevice->aTransmit[1u] = (uint8_t)((address & 0x00FF0000UL) >> 16u);
258  hDevice->aTransmit[2u] = (uint8_t)((address & 0x0000FF00UL) >> 8);
259  hDevice->aTransmit[3u] = (uint8_t) (address & 0x000000FFUL);
260  for (uint32_t i = 0u; i < bufSize; i++)
261  {
262  hDevice->aTransmit[ADI_W25Q32_PROLOGUE + i] = buf[i];
263  }
264 
265  eSpiResult = adi_spi_MasterSubmitBuffer(hDevice->hSpiDevice, &transceive);
266  RETURN_ERROR(eSpiResult, ADI_SPI_SUCCESS, ADI_W25Q32_FAILED_SPI_ERROR);
267  eSpiResult = adi_spi_GetBuffer(hDevice->hSpiDevice, &nHwError);
268  RETURN_ERROR(eSpiResult, ADI_SPI_SUCCESS, ADI_W25Q32_FAILED_SPI_ERROR);
269 
270  /* Wait for write to complete */
271  eResult = busyWait(hDevice);
272  RETURN_ERROR(eResult, ADI_W25Q32_SUCCESS, eResult);
273 
274  return ADI_W25Q32_SUCCESS;
275 }

388  transceive.ReceiverBytes = 0u;
389  transceive.nTxIncrement = 1u;
390  transceive.nRxIncrement = 1u;
391  transceive.bDMA = false;
392  transceive.bRD_CTL = false;
393  transceive.pTransmitter = hDevice->aTransmit;
394  transceive.pReceiver = NULL;
395 
396  /* Fill TX buffer */

409 
410  /* Wait for erase to complete */
411  eResult = busyWait(hDevice);
412  RETURN_ERROR(eResult, ADI_W25Q32_SUCCESS, eResult);
413 
414  return ADI_W25Q32_SUCCESS;

462 static ADI_W25Q32_RESULT command(ADI_W25Q32_HANDLE hDevice, uint8_t nCommand, uint8_t * rxBuffer, uint32_t rxBytes)
463 {
464  ADI_SPI_TRANSCEIVER transceive;
465  ADI_SPI_RESULT eSpiResult;
466  uint8_t txBuffer;
467 
468  transceive.TransmitterBytes = 1u;
469  transceive.ReceiverBytes = (uint16_t) rxBytes;
470  transceive.nTxIncrement = 1u;
471  transceive.nRxIncrement = 1u;
472  transceive.bDMA = false;
473  transceive.pTransmitter = &txBuffer;
474  transceive.pReceiver = rxBuffer;
475  if (rxBytes != 0u)
476  {
477  transceive.bRD_CTL = true;
478  }
479  else
480  {
481  transceive.bRD_CTL = false;
482  }
483 
484  /* Fill TX buffer */
485  txBuffer = nCommand;
486 
487  /* Clear RX buffer */
488  memset(rxBuffer, 0, rxBytes);
489 
490  /* Perform transfer */
491  eSpiResult = adi_spi_MasterReadWrite(hDevice->hSpiDevice, &transceive);
492  RETURN_ERROR(eSpiResult, ADI_SPI_SUCCESS, ADI_W25Q32_FAILED_SPI_ERROR);
493 
494  return ADI_W25Q32_SUCCESS;
495 }

498 static ADI_W25Q32_RESULT busyWait(ADI_W25Q32_HANDLE hDevice)
499 {
500  ADI_W25Q32_RESULT eResult;
501  uint8_t rxBuffer;
502  do
503  {
504  /* Wait for the busy bit to clear */
505  eResult = command(hDevice, ADI_W25Q32_CMD_RDSR, &rxBuffer, 1u);
506  RETURN_ERROR(eResult, ADI_W25Q32_SUCCESS, eResult);
507  } while((rxBuffer & ADI_W25Q32_STAT_BSY) == ADI_W25Q32_STAT_BSY);
508 
509  return ADI_W25Q32_SUCCESS;
510 }

170 ADI_I2C_RESULT adi_i2c_Open (uint32_t const DeviceNum, void* const pMemory, uint32_t const MemorySize, ADI_I2C_HANDLE* const phDevice) {
171 
172  /* make a device handle out of the user memory */
173  ADI_I2C_HANDLE hDevice = (ADI_I2C_HANDLE)pMemory;
174 
175 #if defined(ADI_DEBUG)
176  /* check requested device number */
177  if (DeviceNum >= (uint32_t)ADI_I2C_NUM_INSTANCES) {
178  return ADI_I2C_BAD_DEVICE_NUMBER;
179  }
180 
181  /* verify device is not already open */
182  if (i2c_device_info[DeviceNum].hDevice != NULL) {
183  return ADI_I2C_DEVICE_IN_USE;
184  }
185 
186  /* verify memory size macro value */
187  assert(ADI_I2C_MEMORY_SIZE == sizeof(ADI_I2C_DEV_DATA_TYPE));
188 
189  /* verify user-provided memory meets requirement */
190  if ((NULL == pMemory) || (MemorySize < (uint32_t)ADI_I2C_MEMORY_SIZE)) {
191  return ADI_I2C_INSUFFICIENT_MEMORY;
192  }
193 #endif
194 
195  /* store a bad handle in case of failure */
196  *phDevice = NULL;
197 
198  /*
199  * Link user memory (handle) to ADI_I2C_DEVICE_INFO data structure.
200  *
201  * ADI_I2C_DEVICE_INFO <==> ADI_I2C_HANDLE
202  *
203  * Clear the ADI_I2C_HANDLE memory. This also sets all bool
204  * structure members to false so we do not need to waste cycles
205  * setting these explicitly (e.g. hDevice->bRepearStart = false)
206  */
207  i2c_device_info[DeviceNum].hDevice = (ADI_I2C_DEV_DATA_TYPE *)pMemory;
208  memset(pMemory, 0, MemorySize);
209 
210  /* also link device handle within __ADI_I2C_DEV_DATA_TYPE data structure */
211  hDevice->pDevInfo = &i2c_device_info[DeviceNum];
212  /*
213  * Although the ADI_I2C_DEVICE_INFO struct has the physical device pointer
214  * for this instance, copying it to the ADI_I2C_HANDLE struct (in user memory)
215  * will minimize the runtime footprint and cycle count when accessing the I2C
216  * registers.
217  */
218  hDevice->pDev = i2c_device_info[DeviceNum].pDev;
219 
220  /* store a pointer to user's static configuration settings */
221  hDevice->pDevInfo->pConfig = (ADI_I2C_CONFIG*)&gConfigInfo[DeviceNum];
222 
223  /* create the semaphore */
224  SEM_CREATE(hDevice, "i2c_sem", ADI_I2C_SEMAPHORE_FAILED)
225  ;
226 
227  /* reset the driver and HW state */
228  ADI_I2C_RESULT ignore ADI_UNUSED_ATTRIBUTE = i2cReset(hDevice);
229 
230  /* store device handle into user handle */
231  *phDevice = (ADI_I2C_HANDLE)hDevice;
232 
233  return ADI_I2C_SUCCESS;
234 }

330 ADI_I2C_RESULT adi_i2c_ReadWrite (ADI_I2C_HANDLE const hDevice, ADI_I2C_TRANSACTION* const pTransaction, uint32_t* const pHwErrors) {
331 
332 #ifdef ADI_DEBUG
333  if (IsDeviceHandle(hDevice)) {
334  return ADI_I2C_BAD_DEVICE_HANDLE;
335  }
336  if (I2C_BUSY) {
337  return ADI_I2C_DEVICE_IN_USE;
338  }
339  /* NULL transaction data pointer or zero transaction data count */
340  if ((NULL == pTransaction->pData) || (0u == pTransaction->nDataSize)) {
341  return ADI_I2C_INVALID_PARAMETER;
342  }
343 #endif
344 
345  /* reset submit/get safeguard flag */
346  hDevice->bSubmitCalled = false;
347 
348  /* submit/commence the transaction */
349  submitTransaction(hDevice, pTransaction);
350 
351  /* block on internal transaction completion/error semaphore */
352  if (ADI_I2C_SUCCESS == hDevice->result) {
353 
354  SEM_PEND(hDevice, ADI_I2C_SEMAPHORE_FAILED);
355 
356  /* completion interrupt comes as FIFO unloads, but serialization may not be complete yet... */
357  /* must also wait for hardware busy status to clear before giving back control */
358  /* i.e., allow any transmit serialization to complete after last FIFO unload */
359  while (I2C_BUSY) {
360  ;
361  }
362  }
363 
364  /* copy out any hardware errors... */
365  *pHwErrors = hDevice->hwErrors;
366  if (0u != hDevice->hwErrors) {
367  /* set the HW error return code */
368  hDevice->result = ADI_I2C_HW_ERROR_DETECTED;
369  }
370 
371  /* return transaction result code */
372  return hDevice->result;
373 }

631 ADI_I2C_RESULT adi_i2c_SetBitRate (ADI_I2C_HANDLE const hDevice, uint32_t const requestedBitRate32) {
632 
633  uint32_t clockFrequency32, halfClock32;
634  uint16_t halfClock16;
635  uint16_t highTime16, lowTime16;
636 
637 #ifdef ADI_DEBUG
638  if (IsDeviceHandle(hDevice)) {
639  return ADI_I2C_BAD_DEVICE_HANDLE;
640  }
641  if (I2C_BUSY) {
642  return ADI_I2C_DEVICE_IN_USE;
643  }
644 #endif
645 
646  /* get input clockrate from power service */
647  if (ADI_PWR_SUCCESS != adi_pwr_GetClockFrequency(ADI_CLOCK_PCLK, &clockFrequency32)) {
648  return ADI_I2C_BAD_SYS_CLOCK;
649  }
650 
651  /* block requests above max rated 400kHz operation */
652  if (ADI_I2C_MAX_RATE < requestedBitRate32) {
653  return ADI_I2C_BAD_BITRATE;
654  }
655 
656  /* compute half-cycle period in 32-bits (">>1" is divide by 2) */
657  halfClock32 = (clockFrequency32 / requestedBitRate32) >> 1; /* HRM equation */
658 
659  /* downcast to 16-bit to match destination field */
660  halfClock16 = (uint16_t)(halfClock32 & 0x0000ffffu);
661 
662  /* check for lost precision in conversion */
663  if (halfClock32 != halfClock16) {
664  return ADI_I2C_BAD_BITRATE;
665  }
666 
667  /* adjust high and low durations per HRM */
668  highTime16 = halfClock16 - 7u; /* empirical: varies with board layout, pullups, etc */
669  lowTime16 = halfClock16 - 1u;
670 
671  /* shift values into their clock rate divider register positions */
672  highTime16 <<= BITP_I2C_DIV_HIGH;
673  lowTime16 <<= BITP_I2C_DIV_LOW;
674 
675  /* check for divider overflows beyond designated (8-bit) field masks */
676  if ( (uZero16 != ((uint16_t)highTime16 & (uint16_t)(~(BITM_I2C_DIV_HIGH))))
677  ||
678  (uZero16 != ((uint16_t)lowTime16 & (uint16_t)(~(BITM_I2C_DIV_LOW))))
679  ) {
680  return ADI_I2C_BAD_BITRATE;
681  }
682 
683  /* program new values */
684  hDevice->pDev->DIV = highTime16 | lowTime16;
685 
686  return ADI_I2C_SUCCESS;
687 }

811 static void submitTransaction(ADI_I2C_HANDLE const hDevice, ADI_I2C_TRANSACTION* const pTransaction) {
812 
813  /* reset internal return code */
814  hDevice->result = ADI_I2C_SUCCESS;
815 
816  /* reset hardware error code */
817  hDevice->hwErrors = ADI_I2C_HW_ERROR_NONE;
818 
819  /* wait for HW to be ready */
820  while (I2C_BUSY) {
821  ;
822  }
823 
824  /* save common user parameters */
825  hDevice->pNextPrologueByte = pTransaction->pPrologue;
826  hDevice->remainingPrologueCount = pTransaction->nPrologueSize;
827  hDevice->bRepeatStart = pTransaction->bRepeatStart;
828 
829  /* encode (mask and upshift) the slave address, leaving room for the r/w control bit (LSB) */
830  hDevice->i2cEncodedDeviceAddress = (hDevice->i2cDeviceAddress & (BITM_I2C_ADDR1_VALUE >> 1)) << 1;
831 
832  /* dispatch */
833  if (pTransaction->bReadNotWrite) {
834 
835  /* setup read parameters */
836  hDevice->pNextReadByte = pTransaction->pData;
837  hDevice->remainingReadCount = pTransaction->nDataSize;
838  hDevice->pNextWriteByte = NULL;
839  hDevice->remainingWriteCount = 0u;
840 
841  /* set read bit */
842  hDevice->i2cEncodedDeviceAddress |= READ_NOT_WRITE;
843 
844  /* commence receive */
845  commenceReceive(hDevice);
846 
847  } else {
848 
849  /* setup write parameters */
850  hDevice->pNextReadByte = NULL;
851  hDevice->remainingReadCount = 0u;
852  hDevice->pNextWriteByte = pTransaction->pData;
853  hDevice->remainingWriteCount = pTransaction->nDataSize;
854 
855  /* clear read bit */
856  hDevice->i2cEncodedDeviceAddress &= (~READ_NOT_WRITE);
857 
858  /* commence transmit */
859  commenceTransmit(hDevice);
860  }
861 }

864 static void commenceTransmit(ADI_I2C_HANDLE const hDevice) {
865 
866  /* transmit is always pure transmit, whether we have a prologue or not... */
867 
868  /* enable PIO interrupts */
869  NVIC_EnableIRQ(hDevice->pDevInfo->pioIRQn);
870 
871  /* enable i2c for PIO-based transmit interrupts */
872  hDevice->pDev->MCTL |= (BITM_I2C_MCTL_IENMTX | BITM_I2C_MCTL_MASEN);
873 
874  /* how many bytes are available in the transmit FIFO (2-deep) */
875  uint16_t writableBytes = 2u - (hDevice->pDev->MSTAT & (uint16_t)BITM_I2C_MSTAT_MTXF);
876 
877  /* prime transmit FIFO with any prologue data */
878  while ((0u < writableBytes) && (hDevice->remainingPrologueCount)) {
879  hDevice->pDev->MTX = *hDevice->pNextPrologueByte;
880  hDevice->pNextPrologueByte++;
881  hDevice->remainingPrologueCount--;
882  writableBytes--;
883  }
884 
885  /* flesh out any remaining FIFO space with transmit data */
886  while ((0u < writableBytes) && (hDevice->remainingWriteCount)) {
887  hDevice->pDev->MTX = *hDevice->pNextWriteByte;
888  hDevice->pNextWriteByte++;
889  hDevice->remainingWriteCount--;
890  writableBytes--;
891  }
892 
893  /* launch the transmit */
894  hDevice->pDev->ADDR1 = hDevice->i2cEncodedDeviceAddress;
895 }

898 /* initiate receive addressing phase */
899 static void commenceReceive(ADI_I2C_HANDLE const hDevice) {
900 
901  /* receive can be either pure receive (no prologue),
902  or a transmit (of prologue) followed by a receive */
903 
904  /* enable PIO interrupts */
905  NVIC_EnableIRQ(hDevice->pDevInfo->pioIRQn);
906 
907  /* enable i2c for PIO-based receive interrupts */
908  hDevice->pDev->MCTL |= (uint16_t)(BITM_I2C_MCTL_IENMRX | BITM_I2C_MCTL_MASEN);
909 
910  /* program HW receive count */
911  if (hDevice->remainingReadCount > BITM_I2C_MRXCNT_EXTEND) {
912  hDevice->pDev->MRXCNT = BITM_I2C_MRXCNT_EXTEND;
913  hDevice->remainingReadCount -= BITM_I2C_MRXCNT_EXTEND;
914  } else {
915  hDevice->pDev->MRXCNT = hDevice->remainingReadCount - 1u;
916  hDevice->remainingReadCount = 0u;
917  }
918 
919  /* if we have prologue (the dreaded "COMBINED FORMAT"), transmit the prologue prior to data receive... */
920  if (hDevice->remainingPrologueCount) {
921 
922  /* -OR- in transmit interrupt enable if we have prologue data to send */
923  hDevice->pDev->MCTL |= BITM_I2C_MCTL_IENMTX;
924 
925  /* how many bytes are available in the transmit FIFO (should be 2) */
926  uint16_t writableBytes = 2u - (hDevice->pDev->MSTAT & (uint16_t)BITM_I2C_MSTAT_MTXF);
927 
928  /* prime transmit FIFO with any prologue data (memory address or command) first */
929  while ((0u < writableBytes) && (hDevice->remainingPrologueCount)) {
930  hDevice->pDev->MTX = *hDevice->pNextPrologueByte;
931  hDevice->pNextPrologueByte++;
932  hDevice->remainingPrologueCount--;
933  writableBytes--;
934  }
935 
936  /* initiate prologue transmit with read bit cleared (for prologue write) */
937  /* (read sequence is initiated by transmit handler, *after* prologue is transmitted...) */
938  hDevice->pDev->ADDR1 = hDevice->i2cEncodedDeviceAddress & (uint16_t)(~READ_NOT_WRITE);
939 
940  } else {
941 
942  /* no prologue... initiate pure receive (read bit already set) */
943  hDevice->pDev->ADDR1 = hDevice->i2cEncodedDeviceAddress;
944  }
945 }

948 /* reset the I2C HW */
949 static ADI_I2C_RESULT i2cReset(ADI_I2C_HANDLE const hDevice) {
950 
951  volatile uint16_t temp;
952  /* disable interrupts */
953  NVIC_DisableIRQ(hDevice->pDevInfo->pioIRQn);
954 
955  /* reset any pending interrupts and TX FIFO (W1C) */
956  temp = hDevice->pDev->MSTAT;
957  hDevice->pDev->MSTAT = temp;
958 
959  /* discard any rogue RX FIFO data */
960  while (uZero16 != (hDevice->pDev->STAT & (uint16_t)BITM_I2C_STAT_MRXF)) {
961  volatile uint16_t delme ADI_UNUSED_ATTRIBUTE = hDevice->pDev->MTX;
962  }
963 
964  /* reset i2c control register */
965  hDevice->pDev->MCTL = 0u;
966 
967  /* reset repeat start logic */
968  hDevice->pDev->SHCTL = 1u;
969 
970  /* (re)assert controller defaults from user config values */
971  hDevice->pDev->MCTL = hDevice->pDevInfo->pConfig->MasterControlRegister;
972  hDevice->pDev->DIV = hDevice->pDevInfo->pConfig->ClockDividerRegister;
973  hDevice->pDev->SHCTL = hDevice->pDevInfo->pConfig->SharedControlRegister;
974  hDevice->pDev->TCTL = hDevice->pDevInfo->pConfig->TimingControlRegister;
975  hDevice->pDev->ASTRETCH_SCL = hDevice->pDevInfo->pConfig->ClockStretchRegister;
976  hDevice->i2cDeviceAddress = hDevice->pDevInfo->pConfig->TargetSlaveAddress;
977 
978  return ADI_I2C_SUCCESS;
979 }

987 /* transmit interrupt handler */
988 static void transmitHandler(ADI_I2C_HANDLE const hDevice) {
989 
990  /* how much room in transmit FIFO? */
991  /* DO ***NOT*** USE MSTAT:MTXF... FALSELY INDICATES MOSTLY FULL FIFO! */
992  uint16_t writableBytes = 2u - ((hDevice->pDev->STAT & (uint16_t)BITM_I2C_STAT_MTXF) >> BITP_I2C_STAT_MTXF);
993 
994  /* for extended prologues, continue pushing prologue data out */
995  while ((0u < writableBytes) && (hDevice->remainingPrologueCount)) {
996  hDevice->pDev->MTX = *hDevice->pNextPrologueByte;
997  hDevice->pNextPrologueByte++;
998  hDevice->remainingPrologueCount--;
999  writableBytes--;
1000  }
1001 
1002  /* once the prologue is done... */
1003  if (0u == hDevice->remainingPrologueCount) {
1004 
1005  /* if we have a completed prologue associated with a read sequence... */
1006  if (0u < hDevice->remainingReadCount) {
1007 
1008  /* initiate the read (subsequently driven by receive interrupt handler) */
1009  hDevice->pDev->ADDR1 = hDevice->i2cEncodedDeviceAddress;
1010 
1011  } else {
1012 
1013  /* normal transmit interrupt: just push transmit data */
1014  while ((0u < writableBytes) && (hDevice->remainingWriteCount)) {
1015  hDevice->pDev->MTX = *hDevice->pNextWriteByte;
1016  hDevice->pNextWriteByte++;
1017  hDevice->remainingWriteCount--;
1018  writableBytes--;
1019  }
1020  }
1021  }
1022 
1023  /* clear TX interrupt as we complete transmit writes */
1024  if (0u == hDevice->remainingWriteCount) {
1025  hDevice->pDev->MSTAT = BITM_I2C_MSTAT_MTXREQ;
1026  }
1027 }

1030 /* receive interrupt handler */
1031 static void receiveHandler(ADI_I2C_HANDLE const hDevice) {
1032 
1033  /* note: we never need to deal with prologue data here... it will already be transmitted... */
1034 
1035  /* how many bytes in receive FIFO? */
1036  uint16_t readableBytes = (hDevice->pDev->STAT & (uint16_t)BITM_I2C_STAT_MRXF) >> BITP_I2C_STAT_MRXF;
1037 
1038  /* pull bytes from fifo */
1039  while (0u < readableBytes) {
1040 
1041  readableBytes--;
1042 
1043  /* pull one byte */
1044  *hDevice->pNextReadByte = (uint8_t)hDevice->pDev->MRX;
1045  hDevice->pNextReadByte++;
1046 
1047  if ((0u == hDevice->pDev->MCRXCNT) && (hDevice->remainingReadCount)) {
1048 
1049  /* if HW read counter goes to zero with remaining data to read, reprogram read count */
1050  if (hDevice->remainingReadCount > BITM_I2C_MRXCNT_EXTEND) {
1051  /* use extended count flag for large remaining counts... */
1052  hDevice->pDev->MRXCNT = BITM_I2C_MRXCNT_EXTEND;
1053  hDevice->remainingReadCount -= BITM_I2C_MRXCNT_EXTEND;
1054  } else {
1055  /* new count fits... no need for extended count */
1056  hDevice->pDev->MRXCNT = hDevice->remainingReadCount - 1u;
1057  hDevice->remainingReadCount = 0u;
1058  }
1059  }
1060  }
1061 }
1062 
1063 /* completion interrupt handler */
1064 static void completeHandler(ADI_I2C_HANDLE const hDevice) {
1065 
1066  /* block on busy until all transmit data has both left
1067  the fifo AND has been fully serialized to the bus. */
1068  while (I2C_BUSY) {
1069  ;
1070  }
1071 
1072  /* disable interrupts */
1073  NVIC_DisableIRQ(hDevice->pDevInfo->pioIRQn);
1074 
1075  /* reset controller to default user config state */
1076  hDevice->pDev->MCTL = (uint16_t)gConfigInfo->MasterControlRegister;
1077 }

1080 /* error interrupt handler */
1081 static void errorHandler(ADI_I2C_HANDLE const hDevice) {
1082 
1083  /* accumulate I2C bus errors */
1084 
1085  if (uZero16 != (hDevice->hwStatus & (uint16_t)BITM_I2C_MSTAT_NACKADDR)) {
1086  hDevice->hwErrors |= ADI_I2C_HW_ERROR_NACK_ADDR;
1087  }
1088 
1089  if (uZero16 != (hDevice->hwStatus & (uint16_t)BITM_I2C_MSTAT_NACKDATA)) {
1090  hDevice->hwErrors |= ADI_I2C_HW_ERROR_NACK_DATA;
1091  }
1092 
1093  if (uZero16 != (hDevice->hwStatus & (uint16_t)BITM_I2C_MSTAT_ALOST)) {
1094  hDevice->hwErrors |= ADI_I2C_HW_ERROR_ARBITRATION_LOST;
1095  }
1096 
1097  /* if no other errors exist, note we had an unexpected error */
1098  if (hDevice->hwErrors == ADI_I2C_HW_ERROR_NONE) {
1099  hDevice->hwErrors = ADI_I2C_HW_ERROR_UNEXPECTED_ERROR;
1100  }
1101 }

1109 /* PIO mode I2C interrupt handler */
1110 void I2C0_Master_Int_Handler(void) {
1111 
1112  bool bPost = false;
1113 
1114  /* rtos prologue */
1115  ISR_PROLOG()
1116  ;
1117 
1118  /* recover device handle */
1119  ADI_I2C_HANDLE const hDevice = (ADI_I2C_HANDLE)i2c_device_info[0].hDevice;
1120 
1121  /* save destructive status read... */
1122  hDevice->hwStatus = hDevice->pDev->MSTAT;
1123 
1124  /* if RepeatStart request is pending, rewrite address register ASAP (and only once) to block stop bit */
1125  if (hDevice->bRepeatStart) {
1126  hDevice->pDev->ADDR1 = hDevice->i2cEncodedDeviceAddress;
1127  hDevice->bRepeatStart = false; /* just do it once on 1st interrupt */
1128  }
1129 
1130  /* forward TX interrupts to TX handler */
1131  if (uZero16 != (hDevice->hwStatus & (uint16_t)BITM_I2C_MSTAT_MTXREQ)) {
1132  transmitHandler(hDevice);
1133  }
1134 
1135  /* forward RX interrupts to RX handler */
1136  if (uZero16 != (hDevice->hwStatus & (uint16_t)BITM_I2C_MSTAT_MRXREQ)) {
1137  receiveHandler(hDevice);
1138  }
1139 
1140  /* dispatch any errors */
1141  if (uZero16 != (hDevice->hwStatus & ADI_I2C_STATUS_ERROR_MASK)) {
1142  errorHandler(hDevice);
1143 
1144  /* post on bus error */
1145  bPost = true;
1146  }
1147 
1148  /* transmit complete */
1149  if (uZero16 != (hDevice->hwStatus & BITM_I2C_MSTAT_TCOMP)) {
1150  completeHandler(hDevice);
1151 
1152  /* post on completion */
1153  bPost = true;
1154  }
1155 
1156  /* just post once */
1157  if (true == bPost) {
1158  SEM_POST(hDevice);
1159  }
1160 
1161  /* rtos epilogue */
1162  ISR_EPILOG()
1163  ;
1164 }

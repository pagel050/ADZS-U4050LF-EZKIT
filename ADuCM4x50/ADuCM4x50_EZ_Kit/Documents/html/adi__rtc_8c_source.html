177 ADI_RTC_RESULT adi_rtc_Open(
178  uint32_t DeviceNumber,
179  void *pDeviceMemory,
180  uint32_t MemorySize,
181  ADI_RTC_HANDLE *phDevice
182  )
183 {
184  ADI_RTC_DEVICE *pDevice = pDeviceMemory;
185 
186  /* store a bad handle in case of failure */
187  *phDevice = (ADI_RTC_HANDLE) NULL;
188 
189 #ifdef ADI_DEBUG
190  if ( DeviceNumber >= ADI_RTC_NUM_INSTANCE)
191  {
192  return ADI_RTC_INVALID_INSTANCE;
193  }
194  assert(ADI_RTC_MEMORY_SIZE == sizeof(ADI_RTC_DEVICE));
195  if (aRTCDeviceInfo[DeviceNumber].hDevice != NULL)
196  {
197  return ADI_RTC_IN_USE;
198  }
199  if(MemorySize < ADI_RTC_MEMORY_SIZE)
200  {
201  return(ADI_RTC_FAILURE);
202  }
203 #endif
204 
205  memset(pDeviceMemory,0,MemorySize);
206  /* initialize device data entries */
207  pDevice->pRTCRegs = aRTCDeviceInfo[DeviceNumber].pRTCRegs;
208 
209  PEND_BEFORE_WRITE(SR1,BITM_RTC_SR1_WPNDCR0)
210 
211  pDevice->pRTCRegs->CR0 = 0u;
212  pDevice->pRTCRegs->CR1 = 0u;
213 
214  /* Wait till write to Control Register to take effect */
215  SYNC_AFTER_WRITE(SR0,BITM_RTC_SR0_WSYNCCR0)
216 
217  PEND_BEFORE_WRITE(SR1,BITM_RTC_SR1_WPNDSR0)
218 
219  pDevice->pRTCRegs->SR0 = ADI_RTC_SR3_IRQ_STATUS_MASK;
220  SYNC_AFTER_WRITE(SR0,BITM_RTC_SR0_WSYNCSR0)
221 
222  pDevice->pRTCRegs->CNT0 = 0u;
223  pDevice->pRTCRegs->CNT1 = 0u;
224  SYNC_AFTER_WRITE(SR0,BITM_RTC_SR0_WSYNCCNT0)
225 
226  /* local pointer to instance data */
227  aRTCDeviceInfo[DeviceNumber].hDevice = pDevice;
228  pDevice->pDeviceInfo = &aRTCDeviceInfo[DeviceNumber];
229 
230  /* Use static configuration to initialize the RTC */
231  rtc_init(pDevice,&aRTCConfig[DeviceNumber]);
232 
233  /* store handle at application handle pointer */
234  *phDevice = pDevice;
235  pDevice->eIRQn = aRTCDeviceInfo[DeviceNumber].eIRQn;
236  /* Enable RTC interrupts in NVIC */
237  NVIC_EnableIRQ((IRQn_Type)(pDevice->eIRQn));
238 
239  return ADI_RTC_SUCCESS; /* initialized */
240 }

548 ADI_RTC_RESULT adi_rtc_EnableInterrupts (ADI_RTC_HANDLE const hDevice, ADI_RTC_INT_TYPE Interrupts, bool bEnable)
549 {
550  ADI_RTC_DEVICE *pDevice = hDevice;
551 
552 #ifdef ADI_DEBUG
553  ADI_RTC_RESULT eResult;
554  if((eResult = ValidateHandle(pDevice)) != ADI_RTC_SUCCESS)
555  {
556  return eResult;
557  }
558  if( (pDevice->pRTCRegs == pADI_RTC0) &&(((uint16_t)((ADI_RTC_MOD60ALM_INT | ADI_RTC_ISO_DONE_INT|
559  ADI_RTC_COUNT_INT |
560  ADI_RTC_TRIM_INT | ADI_RTC_COUNT_ROLLOVER_INT |
561  ADI_RTC_MOD60_ROLLOVER_INT
562  )) & (uint16_t)Interrupts) != 0u))
563  {
564  return(ADI_RTC_INVALID_PARAM);
565  }
566 
567  assert(sizeof(Interrupt_Details)/sizeof(Interrupt_Details[0]) == ADI_RTC_NUM_INTERRUPTS);
568 #endif
569 
570  /* TODO - more sync for new registers */
571  PEND_BEFORE_WRITE(SR1,BITM_RTC_SR1_WPNDCR0)
572  PEND_BEFORE_WRITE(SR2,BITM_RTC_SR2_WPNDCR1MIR)
573 
574  uint8_t ndx = 0u;
575  cr0 = 0u; cr1 = 0u; cr3oc = 0u; cr4oc = 0u; cr2ic = 0u; cr5ocs = 0u;
576 
577  while( Interrupts )
578  {
579  if( 0u != (Interrupts & 1u) )
580  {
581  uint16_t *cr = Interrupt_Details[ndx].cr;
582  uint16_t enableBitPosition = Interrupt_Details[ndx].bitPositionl;
583  *cr = *cr | (1u << enableBitPosition);
584  }
585  Interrupts >>= 1;
586  ndx++;
587  }
588  /* set/clear interrupt enable bit(s) in control register */
589  if (bEnable)
590  {
591  pDevice->pRTCRegs->CR0 |= cr0;
592  pDevice->pRTCRegs->CR1 |= cr1;
593  pDevice->pRTCRegs->CR3SS |= cr3oc;
594  pDevice->pRTCRegs->CR4SS |= cr4oc;
595  pDevice->pRTCRegs->CR2IC |= cr2ic;
596  pDevice->pRTCRegs->CR5SSS |= cr5ocs;
597 
598  }
599  else
600  {
601  pDevice->pRTCRegs->CR0 &= ~cr0;
602  pDevice->pRTCRegs->CR1 &= ~cr1;
603  pDevice->pRTCRegs->CR3SS &= ~cr3oc;
604  pDevice->pRTCRegs->CR4SS &= ~cr4oc;
605  pDevice->pRTCRegs->CR2IC &= ~cr2ic;
606  pDevice->pRTCRegs->CR5SSS &= ~cr5ocs;
607  }
608  SYNC_AFTER_WRITE(SR0,BITM_RTC_SR0_WSYNCCR0)
609  SYNC_AFTER_WRITE(SR2,BITM_RTC_SR2_WSYNCCR1MIR)
610  return ADI_RTC_SUCCESS;
611 }

194 ADI_ADXL363_RESULT adi_adxl363_Open(void *pDevMemory,
195  uint32_t nMemorySize,

198  /* Pointer to the ADXL363 device */

200  /* Pointer to the ADXL363 instance information */
201  ADI_ADXL363_INFO *pDevInfo;
202  /* Spi enumeration return */

204  /* This API return */
205  ADI_ADXL363_RESULT eResult = ADI_ADXL363_SUCCESS;
206 
207 #ifdef ADI_DEBUG
208 
209  /* Check if the given memory is sufficient to operate at least in interrupt mode */
210  if(nMemorySize < ADI_ADXL363_MEMORY_SIZE)
211  {
212  return ADI_ADXL363_INSUFFICIENT_MEMORY;
213  }
214 
215  /* Check if the given pointer parameters are valid. */
216  if((pDevMemory == NULL) || (phDevice == NULL))
217  {
218  return ADI_ADXL363_NULL_POINTER;
219  }
220 
221  /* Check if the given device instance is already opened */
222  if(gADXL363Dev[0].eState != ADI_ADXL363_STATE_NOT_OPENED)
223  {
224  return ADI_ADXL363_DEVICE_ALREADY_OPENED;
225  }
226 
227  assert(ADI_ADXL363_MEMORY_SIZE == sizeof(ADI_ADXL363_INFO));
228 #endif /* ADI_DEBUG */
229 
230  pDevice = &gADXL363Dev[0];
231 
232  /* Fragment the given memory for various needs */
233  pDevice->pDevInfo = (ADI_ADXL363_INFO * )pDevMemory;
234  pDevInfo = pDevice->pDevInfo;
235 
236  /* Clear the Device Information structure */
237  memset((void *)pDevice->pDevInfo, 0, sizeof(ADI_ADXL363_INFO));
238 
239  /* Open the SPI */
240  eSpiResult = adi_spi_Open(ADXL363_CFG_SPI_DEV_NUM,
241  pDevInfo->aDeviceMemory,
242  ADI_SPI_MEMORY_SIZE,
243  &(pDevInfo->hSPIDevice));
244 
245  if(eSpiResult == ADI_SPI_SUCCESS) {
246 
247  /* Set the chip select */
248  eSpiResult = adi_spi_SetChipSelect(pDevInfo->hSPIDevice,
249  ADXL363_CFG_SPI_DEV_CS);
250 
251  if(eSpiResult == ADI_SPI_SUCCESS) {
252  /* Mark the device as opened */
253  pDevice->eState = ADI_ADXL363_STATE_OPENED;
254 
255  Init_adxl363(pDevice);
256 
257  /* Mark the device as on Standby */
258  pDevice->eState = ADI_ADXL363_STATE_STANDBY;
259 
260  /* return the handle to the opened device*/
261  *phDevice = pDevice;
262  }
263  else {
264  /* Unable to set the chip select, so close it all out, and return a failure. */
265  eSpiResult = adi_spi_Close(pDevInfo->hSPIDevice);
266  eResult = ADI_ADXL363_SPI_DEV_OPEN_FAILED;
267  }
268  }
269  else {
270  eResult = ADI_ADXL363_SPI_DEV_OPEN_FAILED;
271  }
272 
273  return eResult;
274 }

466 ADI_ADXL363_RESULT adi_adxl363_Enable(
467  ADI_ADXL363_HANDLE hDevice,
468  bool bEnable
469 )
470 {
471  /* Pointer to the ADXL363 Device data */
472  ADI_ADXL363_DEVICE *pDevice = (ADI_ADXL363_DEVICE *) hDevice;
473 
474  /* POWER_CTL register contents */
475  uint8_t PwrCtlData = 0u;
476 
477  /* Pointer to device instance information */
478  ADI_ADXL363_INFO *pDevInfo;
479 
480 #ifdef ADI_DEBUG
481  /* ADXL363 result code */
482  ADI_ADXL363_RESULT eResult;
483 
484  /* Validate the given device handle */
485  if((eResult = ValidateHandle(pDevice)) != ADI_ADXL363_SUCCESS)
486  {
487  return eResult;
488  }
489  /* Check if measurement is already enabled when trying to enable it */
490  if((bEnable == true) && (pDevice->eState == ADI_ADXL363_STATE_MEASURING))
491  {
492  return ADI_ADXL363_INVALID_OPERATION;
493  }
494  /* Check if measurement already disabled when trying to disable it */
495  if((bEnable == false) && (pDevice->eState == ADI_ADXL363_STATE_STANDBY))
496  {
497  return ADI_ADXL363_INVALID_OPERATION;
498  }
499 #endif /* ADI_DEBUG */
500 
501  /* Get the pointer to device instance information */
502  pDevInfo = pDevice->pDevInfo;
503 
504  /* Get the current contents of power control register */
505  ADXL363_REG_READ(ADI_ADXL363_POWER_CTL, &PwrCtlData)
506 
507  if(bEnable == true) {
508  /* DATA_FORMAT register contents */
509  uint8_t DataFormat;
510  /* Edge sense type */
511  ADI_GPIO_IRQ_TRIGGER_CONDITION eSense;
512 
513  /* Read the contents of the Data format register */
514  ADXL363_REG_READ(ADI_ADXL363_INT1_MAP, &DataFormat)
515 
516  /* Choose the GPIO sensing based on the interrupt polarity */
517  eSense = ((DataFormat & BITM_ADXL363_INT_LOW) != 0u) ? \
518  ADI_GPIO_IRQ_FALLING_EDGE : ADI_GPIO_IRQ_RISING_EDGE ;
519 
520  /* If this GPIO line has been setup for use, then register the
521  callback for it. */
522  if(pDevInfo->IntGPIOInfo[0].eIRQGrp != 0u) {
523  if(SetupGPIO(pDevInfo,
524  &pDevInfo->IntGPIOInfo[0],
525  pDevice->pDevInfo->pfCallback,
526  pDevice->pDevInfo->pCBParam,
527  eSense) != ADI_ADXL363_SUCCESS) {
528  return ADI_ADXL363_GPIO_FAILED;
529  }
530  }
531 
532  /* If this GPIO line has been setup for use, then register the
533  callback for it. */
534  if(pDevInfo->IntGPIOInfo[1].eIRQGrp != 0u) {
535  if(SetupGPIO(pDevInfo,
536  &pDevInfo->IntGPIOInfo[1],
537  pDevice->pDevInfo->pfCallback,
538  pDevice->pDevInfo->pCBParam,
539  eSense) != ADI_ADXL363_SUCCESS) {
540  return ADI_ADXL363_GPIO_FAILED;
541  }
542  }
543 
544  /* Set the Measure bit to enable measurement */
545  PwrCtlData |= (2u << BITP_ADXL363_MEASURE_EN);
546 
547  /* Change the state to measuring */
548  pDevice->eState = ADI_ADXL363_STATE_MEASURING;
549  }
550  else {
551  /* Clear the Measure bit to put the part into standby mode */
552  PwrCtlData &= ~(BITM_ADXL363_MEASURE_EN);
553 
554  /* Change the state to standby */
555  pDevice->eState = ADI_ADXL363_STATE_STANDBY;
556  }
557 
558  /* Set the current contents of Power control register */
559  ADXL363_REG_WRITE(ADI_ADXL363_POWER_CTL, PwrCtlData)
560 
561  return ADI_ADXL363_SUCCESS;
562 }

170 ADI_GPIO_RESULT adi_gpio_Init(
171  void* const pMemory,
172  uint32_t const MemorySize
173 )
174 {
175 
176 #ifdef ADI_DEBUG
177  /* Verify the given memory pointer */
178  if(NULL == pMemory)
179  {
180  return ADI_GPIO_NULL_PARAMETER;
181  }
182  /* Check if the memory size is sufficient to operate the driver */
183  if(MemorySize < ADI_GPIO_MEMORY_SIZE)
184  {
185  return ADI_GPIO_INVALID_MEMORY_SIZE;
186  }
187  assert(ADI_GPIO_MEMORY_SIZE == sizeof(ADI_GPIO_DEV_DATA));
188 #endif
189 
190  /* Only initialize on 1st init call, i.e., preserve callbacks on multiple inits */
191  if (NULL == adi_gpio_Device.pData)
192  {
193  uint32_t i;
194 
195  adi_gpio_Device.pData = (ADI_GPIO_DEV_DATA*)pMemory;
196 
197  /* Initialize the callback table */
198  for (i = 0u; i < ADI_GPIO_NUM_INTERRUPTS; i++)
199  {
200  adi_gpio_Device.pData->CallbackTable[i].pfCallback = NULL;
201  adi_gpio_Device.pData->CallbackTable[i].pCBParam = NULL;
202  }
203 
204  /* Enable the group interrupts */
205  NVIC_EnableIRQ(SYS_GPIO_INTA_IRQn);
206  NVIC_EnableIRQ(SYS_GPIO_INTB_IRQn);
207  }
208 
209  return (ADI_GPIO_SUCCESS);
210 }

263 ADI_GPIO_RESULT adi_gpio_SetGroupInterruptPins(const ADI_GPIO_PORT Port, const ADI_GPIO_IRQ eIrq, const ADI_GPIO_DATA Pins)
264 {
265  ADI_GPIO_TypeDef *pPort; /* pointer to port registers */
266  ADI_INT_STATUS_ALLOC();
267 #ifdef ADI_DEBUG
268  /* make sure we're initialized */
269  if (NULL == adi_gpio_Device.pData)
270  {
271  return (ADI_GPIO_NOT_INITIALIZED);
272  }
273 
274  /* validate the pins */
275  if (!ArePinsValid(Port, Pins))
276  {
277  return (ADI_GPIO_INVALID_PINS);
278  }
279 #endif
280 
281  pPort = adi_gpio_Device.pReg[Port];
282 
283  ADI_ENTER_CRITICAL_REGION();
284  switch (eIrq)
285  {
286  case SYS_GPIO_INTA_IRQn:
287  pPort->IENA = Pins;
288  break;
289  case SYS_GPIO_INTB_IRQn:

296 
297  return (ADI_GPIO_SUCCESS);

866 /* Common group (A/B) interrupt handler */
867 static void CommonInterruptHandler(const ADI_GPIO_IRQ_INDEX index, const IRQn_Type eIrq)
868 {
869  ADI_GPIO_PORT Port;
870  ADI_GPIO_TypeDef *pPort;
871  ADI_GPIO_DATA Pins;
872  ADI_GPIO_DATA nIntEnabledPins;
873 
874  ADI_GPIO_CALLBACK_INFO *pCallbackInfo = &adi_gpio_Device.pData->CallbackTable[index];
875 
876  /* Loop over all the ports. */
877  for(Port=ADI_GPIO_PORT0; Port<ADI_GPIO_NUM_PORTS; Port++)
878  {
879  pPort = adi_gpio_Device.pReg[Port];
880 
881  /* Is the interrupt is for GROUP A */
882  if(SYS_GPIO_INTA_IRQn == eIrq)
883  {
884  nIntEnabledPins = pPort->IENA;
885  }
886  else /* Is the interrupt is for GROUP B */
887  {
888  nIntEnabledPins = pPort->IENB;
889  }
890 
891  /* Clear only required interrupts */
892  Pins = ((pPort->INT) & nIntEnabledPins);
893  pPort->INT = Pins;
894 
895  /* params list is: application-registered cbParam, Port number, and interrupt status */
896  if((pCallbackInfo->pfCallback != NULL) && (Pins != 0u))
897  {
898  pCallbackInfo->pfCallback (pCallbackInfo->pCBParam, (uint32_t)Port, &Pins);
899  }
900  }
901 }
